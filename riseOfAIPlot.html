Dataset Setup

data <- read_csv(here("data", "trendsInLLMs.csv")) |>
  janitor::clean_names() |>
  select(system, publication_date, training_compute_flop, domain) |>
  drop_na() |> 
  mutate(training_compute_flop = as.numeric(training_compute_flop), 
         publication_date = mdy(publication_date),
         publication_date = if_else(publication_date > today(), publication_date - years(100), publication_date),
    domain = case_when(
    domain %in% c("Vision", "Games", "Drawing", "Language") ~ domain,
    domain %in% "Speech" ~ "Language",
    TRUE ~ "Other"
  ),
  domain = factor(domain) |> fct_relevel(c("Vision", "Games", "Drawing", "Language", "Other"))
  )
